    '5' : ['3', '7'],
    '3' : ['2', '4'],
    '7' : ['8'],
    '2' : [],
    '4' : ['8'],
    '8' : []
}
visited =[]


def dfsp(visited):
    for i in visited:
        print(i,'->',end='')


def dfs(visited, node, final):
    if node not in visited:
        visited.extend(node)
        print(node , ' final ', final)
        if node == final:
            print('found')
            return visited
        for i in graph[node]:
            dfs(visited, i, final)
